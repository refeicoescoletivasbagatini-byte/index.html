<script>

function salvar(){
 let nome = document.getElementById("nome").value;

 let dados = {
  telefone: telefone.value,
  queixa: queixa.value,
  refracao: refracao.value,
  diag: diag.value
 };

 localStorage.setItem(nome, JSON.stringify(dados));
 alert("Salvo!");
 listar();
}

function listar(){
 let lista = document.getElementById("lista");
 lista.innerHTML="";

 for(let i=0;i<localStorage.length;i++){
  let nome = localStorage.key(i);

  lista.innerHTML += `<div class="lista" onclick="abrir('${nome}')">${nome}</div>`;
 }
}

function abrir(nome){
 let d = JSON.parse(localStorage.getItem(nome));

 document.getElementById("nome").value = nome;
 telefone.value = d.telefone;
 queixa.value = d.queixa;
 refracao.value = d.refracao;
 diag.value = d.diag;
}

function buscar(txt){
 let lista = document.getElementById("lista");
 lista.innerHTML="";

 for(let i=0;i<localStorage.length;i++){
  let nome = localStorage.key(i);

  if(nome.toLowerCase().includes(txt.toLowerCase())){
   lista.innerHTML += `<div class="lista" onclick="abrir('${nome}')">${nome}</div>`;
  }
 }
}

function novo(){
 document.querySelectorAll("input,textarea").forEach(e=>e.value="");
}

async function gerarPDF(){
 const { jsPDF } = window.jspdf;
 const doc = new jsPDF();

 doc.text("Jullyana Bagatini - Optometrista", 20,20);
 doc.text("Paciente: " + nome.value, 20,40);
 doc.text("Queixa: " + queixa.value, 20,60);
 doc.text("Refração: " + refracao.value, 20,80);
 doc.text("Diagnóstico: " + diag.value, 20,100);

 doc.save(nome.value + ".pdf");
}

function whats(){
 let msg = `Paciente: ${nome.value}\nRefração: ${refracao.value}\nDiagnóstico: ${diag.value}`;
 window.open("https://wa.me/5551999434629?text="+encodeURIComponent(msg));
}

listar();
